def add_rule(*args, &block)
  @last_rule = target = args.first
  case target
  when Hash
    @last_rule = target.keys.first
    rules.merge! target
  when String
    rules[target] = args[1]
  when Regexp
    if block_given?
      rules[target] = block
    else
      rules[target] = args[1]
    end
  end
end
